        function initializeAnimations() {
            // Animación de aparición gradual de secciones
            const observer = new IntersectionObserver((entries) => {
                entries.forEach((entry, index) => {
                    if (entry.isIntersecting) {
                        setTimeout(() => {
                            entry.target.style.opacity = '1';
                            entry.target.style.transform = 'translateY(0)';
                        }, index * 100);
                    }
                });
            }, { threshold: 0.1 });

            // Observar todas las secciones
            const sections = document.querySelectorAll('section');
            sections.forEach((section) => {
                section.style.opacity = '0';
                section.style.transform = 'translateY(30px)';
                section.style.transition = 'opacity 0.8s ease, transform 0.8s ease';
                observer.observe(section);
            });

            // Efectos hover adicionales
            const serviceCards = document.querySelectorAll('.hover\\:bg-opacity-20');
            serviceCards.forEach(card => {
                card.addEventListener('mouseenter', () => {
                    card.style.boxShadow = '0 20px 40px rgba(0, 212, 255, 0.3)';
                });
                card.addEventListener('mouseleave', () => {
                    card.style.boxShadow = '';
                });
            });

            // Cursor personalizado con efectos de partículas
            let mouseX = 0, mouseY = 0;
            document.addEventListener('mousemove', (e) => {
                mouseX = e.clientX;
                mouseY = e.clientY;
                
                // Crear partículas ocasionales
                if (Math.random() < 0.1) {
                    createParticle(mouseX, mouseY);
                }
            });
        }

                particle.style.transform = 'scale(0) translateY(-50px)';
            }, 50);
            
            setTimeout(() => {
                particle.remove();
            }, 1000);

        // Función para traducir la página
        function translatePage(language) {
            const elements = document.querySelectorAll('[data-translate]');
            elements.forEach(element => {
                const key = element.getAttribute('data-translate');
                if (translations[language] && translations[language][key]) {
                    element.innerHTML = translations[language][key];
                }
            });
            
            // Actualizar el texto del botón de idioma
            const languageBtn = document.getElementById('language-text');
            if (languageBtn) {
                languageBtn.textContent = translations[language]['language-btn'];
            }
        }
        
        // Función para cambiar idioma
        function toggleLanguage() {
            // Ciclo: español -> inglés -> francés -> español
            switch (currentLanguage) {
                case 'es':
                    currentLanguage = 'en';
                    showLanguageNotification('Changing to English... / Cambiando a inglés...');
                    break;
                case 'en':
                    currentLanguage = 'fr';
                    showLanguageNotification('Changing to French... / Cambiando a francés...');
                    break;
                case 'fr':
                    currentLanguage = 'es';
                    showLanguageNotification('Cambiando a español... / Changing to Spanish...');
                    break;
            }
            
            // Aplicar traducción
            translatePage(currentLanguage);
        }

        function showLanguageNotification(message) {
            const notification = document.createElement('div');
            notification.className = 'fixed top-4 left-1/2 transform -translate-x-1/2 bg-white text-gray-800 px-6 py-3 rounded-full shadow-lg z-50 transition-all duration-300';
            notification.textContent = message;
            document.body.appendChild(notification);
            
            setTimeout(() => {
                notification.style.opacity = '0';
                notification.style.transform = 'translate(-50%, -100%)';
                setTimeout(() => notification.remove(), 300);
            }, 2000);
        }

        // Funcionalidad del buscador de experiencias
        function handleSearchExperiences() {
            // Obtener valores de los selects
            const destino = document.querySelector('select').value;
            const tipoExperiencia = document.querySelectorAll('select')[1].value;
            const duracion = document.querySelectorAll('select')[2].value;
            const presupuesto = document.querySelectorAll('select')[3].value;

            // Validar que se haya seleccionado al menos el destino
            if (destino === 'Selecciona un destino') {
                alert('Por favor selecciona al menos un destino para continuar');
                return;
            }

            // Construir mensaje para WhatsApp
            let mensaje = `🎯 *SOLICITUD DE COTIZACIÓN - MÉXICO MÁGICO*\n\n`;
            mensaje += `Hola! Me interesa obtener información y cotización para el siguiente viaje:\n\n`;
            
            mensaje += `📍 *DESTINO:*\n${destino}\n\n`;
            
            if (tipoExperiencia !== 'Tipo de viaje') {
                mensaje += `🎒 *TIPO DE EXPERIENCIA:*\n${tipoExperiencia}\n\n`;
            }
            
            if (duracion !== 'Días de viaje') {
                mensaje += `📅 *DURACIÓN:*\n${duracion}\n\n`;
            }
            
            if (presupuesto !== 'Rango de precio') {
                mensaje += `💰 *PRESUPUESTO POR PERSONA:*\n${presupuesto}\n\n`;
            }

            mensaje += `📞 Por favor proporciónenme más información y opciones disponibles.\n\n`;
            mensaje += `¡Gracias por su atención!`;

            // Número de WhatsApp de la agencia
            const numeroWhatsApp = '5299993254071'; // Formato: código país + número sin espacios ni signos
            
            // Codificar el mensaje para URL
            const mensajeCodificado = encodeURIComponent(mensaje);
            
            // Crear URL de WhatsApp
            const urlWhatsApp = `https://wa.me/${numeroWhatsApp}?text=${mensajeCodificado}`;
            
            // Abrir WhatsApp
            window.open(urlWhatsApp, '_blank');
        }
